    function formatearFechaNoticia(fechaStr) {

      if (!fechaStr) return '';

      try {

        const fecha = new Date(fechaStr), ahora = new Date(), diffMs = ahora - fecha;

        const diffMins = Math.floor(diffMs / 60000), diffHours = Math.floor(diffMins / 60), diffDays = Math.floor(diffHours / 24);

        if (diffMins < 1) return '🔴 Ahora';

        if (diffMins < 60) return 'Hace ' + diffMins + 'm';

        if (diffHours < 24) return 'Hace ' + diffHours + 'h';

        if (diffDays < 7) return 'Hace ' + diffDays + 'd';

        return fecha.toLocaleDateString('es-AR', { day: '2-digit', month: 'short' });

      } catch(e) { return ''; }

    }

    function renderTablaPersonalizada(data, containerId, tipo) {

      const c = document.getElementById(containerId);

      if (!data || !data.encabezados || !data.datos || !data.datos.length) {

        c.innerHTML = '<div class="empty-state"><div class="empty-icon">🏆</div><div class="empty-title">Sin datos de ' + tipo + '</div><div class="empty-subtitle">Aún no se han registrado partidos en esta etapa</div></div>';

        return;

      }

      const enc = data.encabezados;

      function idx(nombre) {

        return enc.findIndex(function(e) { return e.toString().toUpperCase() === nombre; });

      }

      const iE = idx('EQUIPO'), iSc = idx('ESCUDO'), iPts = idx('PTS'), iPJ = idx('PJ'), iPG = idx('PG'), iPE = idx('PE'), iPP = idx('PP'), iGF = idx('GF'), iGC = idx('GC'), iDG = idx('DG');

      const iF = [];

      for (let i = 1; i <= 6; i++) {

        const x = idx('F' + i);

        if (x >= 0) iF.push(x);

      }

      let h = '<table class="positions-table"><thead><tr><th>#</th><th>Equipo</th><th>PTS</th><th>PJ</th><th>G</th><th>E</th><th>P</th><th>GF</th><th>GC</th><th>DG</th>' + (iF.length ? '<th>Forma</th>' : '') + '</tr></thead><tbody>';

      data.datos.forEach(function(fila, i) {

        const eq = iE >= 0 ? fila[iE] || '' : '';

        const sc = iSc >= 0 ? fila[iSc] || '' : '';

        const pts = iPts >= 0 ? fila[iPts] || 0 : 0;

        const pj = iPJ >= 0 ? fila[iPJ] || 0 : 0;

        const pg = iPG >= 0 ? fila[iPG] || 0 : 0;

        const pe = iPE >= 0 ? fila[iPE] || 0 : 0;

        const pp = iPP >= 0 ? fila[iPP] || 0 : 0;

        const gf = iGF >= 0 ? fila[iGF] || 0 : 0;

        const gc = iGC >= 0 ? fila[iGC] || 0 : 0;

        const dg = iDG >= 0 ? fila[iDG] || 0 : 0;

        const pc = i === 0 ? 'pos-1' : i === 1 ? 'pos-2' : i === 2 ? 'pos-3' : i === 3 ? 'pos-4' : '';

        let fh = '';

        if (iF.length) {

          fh = '<div class="form-cell">';

          iF.forEach(function(fi) {

            const v = (fila[fi] || '').toString().trim().toUpperCase();

            if (v === 'G' || v === 'E' || v === 'P') fh += '<span class="form-badge form-' + v + '">' + v + '</span>';

          });

          fh += '</div>';

        }

        const rowId = 'team-row-' + tipo + '-' + i;

        h += '<tr class="team-row ' + pc + '" onclick="toggleHistorial(\'' + rowId + '\',\'' + eq.replace(/'/g, "\\'") + '\')">';

        h += '<td class="pos-cell">' + (i + 1) + '</td>';

        h += '<td><div class="team-cell">' + (sc ? '<img src="' + sc + '">' : '') + '<span>' + eq + '</span><span class="expand-icon">▼</span></div></td>';

        h += '<td class="pts-cell">' + pts + '</td><td>' + pj + '</td><td>' + pg + '</td><td>' + pe + '</td><td>' + pp + '</td><td>' + gf + '</td><td>' + gc + '</td><td>' + dg + '</td>';

        h += (iF.length ? '<td>' + fh + '</td>' : '');

        h += '</tr>';

        h += '<tr class="historial-row" id="' + rowId + '"><td colspan="' + (iF.length ? 11 : 10) + '" class="historial-cell"><div class="historial-content"><div class="historial-loading">Cargando historial...</div></div></td></tr>';

      });

      h += '</tbody></table>';

      c.innerHTML = h;

    }

    function calcularPosicionesLocal(jugadores) {

      const posiciones = [];

      const arqueros = jugadores.filter(function(j) { return j.posicion === 'ARQ'; });

      const defensas = jugadores.filter(function(j) { return j.posicion === 'DEF'; });

      const mediocampistas = jugadores.filter(function(j) { return j.posicion === 'MED'; });

      const delanteros = jugadores.filter(function(j) { return j.posicion === 'DEL'; });

      arqueros.forEach(function(arq) { posiciones.push({ jugador: arq, x: 5, y: 50 }); });

      if (defensas.length > 0) {

        const espacio = 100 / (defensas.length + 1);

        defensas.forEach(function(def, idx) { posiciones.push({ jugador: def, x: 15, y: espacio * (idx + 1) }); });

      }

      if (mediocampistas.length > 0) {

        const espacio = 100 / (mediocampistas.length + 1);

        mediocampistas.forEach(function(med, idx) { posiciones.push({ jugador: med, x: 28, y: espacio * (idx + 1) }); });

      }

      if (delanteros.length > 0) {

        const espacio = 100 / (delanteros.length + 1);

        delanteros.forEach(function(del, idx) { posiciones.push({ jugador: del, x: 42, y: espacio * (idx + 1) }); });

      }

      return posiciones;

    }

      function crearBarra(label, valorLocal, valorVisitante, esPorcentaje) {

        const total = valorLocal + valorVisitante;

        const pctLocal = total > 0 ? (valorLocal / total * 100) : 50;

        const pctVisitante = total > 0 ? (valorVisitante / total * 100) : 50;

        return '<div class="stat-bar-container"><div class="stat-bar-header"><span class="stat-bar-value">' + valorLocal + (esPorcentaje ? '%' : '') + '</span><span class="stat-bar-label">' + label + '</span><span class="stat-bar-value">' + valorVisitante + (esPorcentaje ? '%' : '') + '</span></div><div class="stat-bar"><div class="stat-bar-local" style="width:' + pctLocal + '%"></div><div class="stat-bar-visitante" style="width:' + pctVisitante + '%"></div></div></div>';

      }

      html += crearBarra('Posesión', localStats.posesion || 0, visitanteStats.posesion || 0, true);

      html += '<div class="stat-section-title">🎯 ATAQUE</div>';

      html += crearBarra('Tiros totales', localStats.tirosTotales || 0, visitanteStats.tirosTotales || 0, false);

      html += crearBarra('Tiros a puerta', localStats.tirosPuerta || 0, visitanteStats.tirosPuerta || 0, false);

      html += crearBarra('Tiros fuera', localStats.tirosFuera || 0, visitanteStats.tirosFuera || 0, false);

      html += crearBarra('Corners', localStats.corners || 0, visitanteStats.corners || 0, false);

      html += crearBarra('Fuera de juego', localStats.fueraJuego || 0, visitanteStats.fueraJuego || 0, false);

      html += '<div class="stat-section-title">🛡️ DEFENSA</div>';

      html += crearBarra('Faltas', localStats.faltas || 0, visitanteStats.faltas || 0, false);

      html += crearBarra('Entradas', localStats.entradas || 0, visitanteStats.entradas || 0, false);

      html += crearBarra('Paradas del portero', localStats.paradas || 0, visitanteStats.paradas || 0, false);

      html += '<div class="stat-section-title">🟨 DISCIPLINA</div>';

      html += crearBarra('Tarjetas amarillas', localStats.amarillas || 0, visitanteStats.amarillas || 0, false);

      html += crearBarra('Tarjetas rojas', localStats.rojas || 0, visitanteStats.rojas || 0, false);

      html += crearBarra('Pases totales', localStats.pases || 0, visitanteStats.pases || 0, false);

      const precisionVisitante = visitanteStats.pases > 0 ? Math.round((visitanteStats.pasesCompletados || 0) / visitanteStats.pases * 100) : 0;

      html += crearBarra('Precisión de pases', precisionLocal, precisionVisitante, true);

      document.getElementById('estadisticasContenido').innerHTML = html;

    }
